function getPool(mode) { // 1. 숫자를 담을 빈 바구니 준비 var all = []; // 2. 사장님 데이터(entireData)에서 숫자만 하나씩 꺼내기 for (var i = 0; i < entireData.length; i++) { var row = entireData[i]; // 첫 번째(회차) 빼고 1번부터 6번 숫자까지 담기 for (var j = 1; j < row.length; j++) { all.push(row[j]); } } // 3. 빈도수 계산 var freq = {}; for (var k = 0; k < all.length; k++) { var num = all[k]; freq[num] = (freq[num] || 0) + 1; } // 4. 많이 나온 순서로 정렬 (사장님 5조/10조 로직 적용) var sorted = Object.keys(freq).sort(function(a, b) { return freq[b] - freq[a]; }).map(Number); // 5. 모드에 맞춰 상위 번호 전달 (5조합은 15개, 10조합은 25개) return sorted.slice(0, mode === '5' ? 15 : 25); } function getCombo(p) { var res = []; var temp = p.slice(); // 복사본 생성 // 6개 무작위 추출 for (var i = 0; i < 6; i++) { var idx = Math.floor(Math.random() * temp.length); res.push(temp.splice(idx, 1)[0]); } // 오름차순 정렬 및 '01' 형태로 변환 return res.sort(function(a, b) { return a - b; }).map(function(n) { return n < 10 ? '0' + n : n.toString(); }); }
